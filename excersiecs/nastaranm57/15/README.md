```
a=imread('C:\Users\PC\Desktop\pool.png');   % وارد کردن تصویر ورودی
b=rgb2gray(a);  % تبدیل آن به تصویر خاکستری
mask=roipoly(b); % ساخت ماسک با استفاده از تابع roipoly
imshow(mask);
j=regionfill(b,mask);
figure;
imshow(j);
```
